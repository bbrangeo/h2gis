Spatial query sample

{% highlight mysql %} -- Count White Walkers 500m around "The Wall" SELECT COUNT(*) AS NB_WHITE_WALKERS FROM CREATURES, LOCATIONS LOC WHERE ST_INTERSECTS(ST_BUFFER(LOC.THE_GEOM, 500), CREATURES.POSITION) AND LOC.NAME = 'THE WALL' AND CREATURES.NAME = 'WHITE WALKERS'; {% endhighlight %}
